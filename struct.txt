#include <stdio.h>
#include <stdlib.h>
 
typedef struct Livro{
  char titulo[30];
  int ano_de_edicao;
  int nro_paginas;
  float preco;
}L;
 
 
int main() {
   L livros[5];
   int i=0, total_paginas=0;
   double media_paginas;
   for(i=0;i<5;i++) {
      printf("Informe o titulo: ") ;
      scanf("%s", &livros[i].titulo);
      printf("Informe o ano de edicao: ");
      scanf("%d", &livros[i].ano_de_edicao);
      printf("Informe o numero de paginas: "); 
      scanf("%d", &livros[i].nro_paginas);
      printf("Informe o preco: "); 
      scanf("%f", &livros[i].preco);
      total_paginas += livros[i].nro_paginas;
   }
    media_paginas = total_paginas/5;
    printf("Media das paginas: %lf",media_paginas);

    return 0;
}








#include <stdio.h>
#include <stdlib.h>

typedef struct Hortaliça {
    char nome[30];
    int ano_colheita;
    int quantidade;
    float preco;
} H;

int main() {
    H hortalicas[2];  // Array para armazenar duas hortaliças
    float soma_precos, diferenca_precos, multiplicacao_precos, divisao_precos;

    for (int i = 0; i < 2; i++) {
        printf("\nInforme o nome da hortaliça %d: ", i + 1);
        scanf("%s", hortalicas[i].nome);

        printf("Informe o ano da colheita: ");
        scanf("%d", &hortalicas[i].ano_colheita);

        printf("Informe a quantidade colhida: ");
        scanf("%d", &hortalicas[i].quantidade);

        printf("Informe o preço por unidade: ");
        scanf("%f", &hortalicas[i].preco);
    }

    // Realizando as operações matemáticas
    soma_precos = hortalicas[0].preco + hortalicas[1].preco;  // Soma dos preços
    diferenca_precos = hortalicas[0].preco - hortalicas[1].preco; // Diferença dos preços
    multiplicacao_precos = hortalicas[0].preco * hortalicas[1].preco; // Multiplicação dos preços
    divisao_precos = hortalicas[0].preco / hortalicas[1].preco; // Divisão dos preços

    // Exibindo os resultados das operações
    printf("\nResultados das operações matemáticas:\n");
    printf("A soma dos preços das hortaliças é: %.2f\n", soma_precos);
    printf("A diferença entre os preços das hortaliças é: %.2f\n", diferenca_precos);
    printf("A multiplicação dos preços das hortaliças é: %.2f\n", multiplicacao_precos);
    printf("A divisão da hortaliça é: %.2f\n", divisao_precos);

    return 0;
}
